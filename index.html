import tkinter as tk # Criando a janela principal janela = tk.Tk() janela.title("Minha IA") janela.geometry("400x200") # largura x altura # Adicionando um rótulo (label) rotulo = tk.Label(janela, text="Olá! Sou sua IA", font=("Arial", 16)) rotulo.pack(pady=20) # adiciona espaço em cima e embaixo # Adicionando um botão def falar_com_ia(): rotulo.config(text="Você clicou no botão!") botao = tk.Button(janela, text="Falar com IA", command=falar_com_ia) botao.pack() # Rodando a janela janela.mainloop()
